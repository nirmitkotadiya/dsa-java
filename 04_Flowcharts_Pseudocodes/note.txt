// Find if number is Prime

// Pseudo Code

1. Start
2. Input N
3. for i = 2 to N-1 do
      if N % i == 0 do
         print "Not Prime"
         Exit
   print "Prime" 
4. Exit

OR 

1. Start
2. Input N
3. let div = 2
4. while div < N do
      if N % div == 0 do
         print "Not Prime"
         Exit
      else
         div = div + 1
   print "Prime"
5. Exit


// Sum of First n natural numbers

1. Start
2. Input N 
3. Let val = 1 & sum = 0 
4. while val <= N  do
      sum = sum + val 
      val = val + 1
5. print sum
6. Exit

OR

1. Start
